SELECT d.depNO,d.DNAME,e.EMPNO,e.ENAME
FROM EMP e
INNER JOIN dept d
ON(e.depno = d.depno)
WHERE EXISTS(select 1 from EMP t where t.mgr = e.empNO)
AND d.depNO in(10,20,30)


SELECT distinct d.depNO,d.DNAME,e2.EMPNO,e2.ENAME
FROM EMP e
INNER JOIN dept d
ON(e.depno = d.depno)
INNER JOIN EMP e2
ON e2.EMPNO = e.mgr
where d.depNO in(10,20,30)
