SELECT e.empno,
       e.ename,
       e.sal,
       d.deptno,
       d.dname,
       d.loc
  FROM Deptno d
  JOIN (SELECT e.*,
               row_number() over (
                 PARTITION BY e.deptno
                     ORDER BY sal DESC) AS rn
          FROM Employee e) e
    ON e.deptno = d.deptno
   AND e.rn = 1
